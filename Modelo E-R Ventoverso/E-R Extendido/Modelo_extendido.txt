table cliente {
	rut_cliente varchar(10) [pk]
	dv_cliente char
	nombre varchar(20)
	apellido varchar(30)
	email varchar(30)
	direccion varchar(50)
	ciudad varchar(50)
	comuna varchar(50)
	region varchar(50)
	password varchar(20)
	telefono varchar(20)
	
}

table pedido {
  id integer [pk]
  rut_cliente varchar(10) [ref: > cliente.rut_cliente]
  id_producto integer [ref: > producto.id]
  estado varchar(10)
  sub_total integer
  total integer
  direccionEnvio varchar(50)
  cantidad integer
  fecha datetime
 
  
}
table producto {
  id integer [pk]
  id_categoria integer [ref: > categoria.id]
  id_subcategoria integer [ref: - subcategoria.id]
  nombre varchar(100)
  descripcion varchar(250)
  precio integer
  stock integer
  imagen blob
  url_producto varchar(250)
}

table detalle_pedido {
id_producto integer [ref: > producto.id]
id_pedido integer [ ref: > pedido.id]

}

table categoria {
id integer [pk]
nombre varchar(10) //nombre categoria
descripcion varchar(250)
}


table subcategoria {
id integer [pk]
nombre varchar(250)
descripcion varchar(250)
id_categoria integer [ref: - categoria.id]

}


table marcas {
  id integer [pk]
  marca varchar(20)
  detalle varchar(250)
}


table categoria_marcas {
  id_categoria integer [ ref: > categoria.id ]
  id_marcas integer [ref: > marcas.id]

}

table carrito {
 id integer [pk]
 rut_cliente varchar(10) [ref: > cliente.rut_cliente]
 status_carrito varchar(50)
 creacion_date datetime
}

table coment_byn {
  id integer [pk]
  rut_cliente varchar(10) [ref: > cliente.rut_cliente]
  id_articulo integer [ref: > articulobyn.id]
  publicacion_date datetime
  contenido varchar(250)

}

table articulobyn {
id integer [pk]
rut_cliente varchar(10) [ref: > cliente.rut_cliente]
titulo varchar(50)
contenido varchar(250)
publicacion_date datetime
autor varchar(50)
}

table carrusel {
id integer [pk]
nombre varchar(50) // Nombre que describe el carrusel
descripcion varchar(250)
fechainicio datetime
fechafin datetime
}

table carrusel_producto {
id_producto integer [ref: > producto.id]
id_carrusel integer [ref: > carrusel.id]
}

table destacado {
id integer [pk]
FechaInicio datetime    //fecha inicio producto destacado        
FechaFin    datetime    //fecha termino producto destacado
estrellas   integer
descuento integer          

}

table producto_destacado {
id_destacado integer [ref: > destacado.id]
id_producto integer [ref: > producto.id]
FechaInicio datetime    //fecha inicio producto destacado        
FechaFin    datetime    //fecha termino producto destacado
}



table metodoEnvio {
id integer [pk]
nombre varchar(20)
descripcion varchar(50)
costo_envio integer
}


table regionEnvio {
id integer [pk]
nombre varchar(20)
codigo_postal integer
}

table envio {
id integer [pk]
id_pedido integer [ref: - pedido.id]
id_metodoEnvio integer [ref: - metodoEnvio.id]
id_regionEnvio integer [ref: > regionEnvio.id]
fc_envio date
}

table metodoPago {
id integer [pk]
nombre_metodo_pago varchar(50)
detalle_metodo_pago varchar(50)
}


table pago {
id integer [pk]
id_pedido integer [ref: - pedido.id]
id_metodoPago integer [ref: - metodoPago.id]
fcPago date
estado varchar(20)
monto integer

}
