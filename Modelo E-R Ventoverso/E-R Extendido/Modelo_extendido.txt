table usuario {
	rut_usuario int [pk]
	dv_usuario char
	nombre varchar[250]
	apellido varchar[250]
	email varchar[250]
	direccion varchar[250]
	ciudad varchar[250]
	comuna varchar[250]
	region varchar[50]
	password varchar[250]
	telefono varchar[250]
	
}

table pedido {
  id int [pk]
  rut_usuario integer [ref: > usuario.rut_usuario]
  id_producto integer [ref: > producto.id]
  estado varchar[10]
  sub_total integer
  total int
  direccionEnvio varchar[250]
  cantidad integer
  fecha datetime
  id_metodoPago integer [ref: > metodo_pago.id]
  
}
table producto {
  id int [pk]
  id_categoria integer [ref: > categoria.id]
  id_subcategoria integer [ref: - subcategoria.id]
  nombre varchar[250]
  descripcion varchar[250]
  precio integer
  stock integer
  imagen varchar[250]
  url_producto varchar[250]
}

table detalle_pedido {
id_producto integer [ref: > producto.id]
id_pedido integer [ ref: > pedido.id]


}

table categoria {
id integer [pk]
nombre string //nombre categoria
descripcion varchar[250]
}


table subcategoria {
id integer [pk]
nombre varchar[250]
descripcion varchar[250]
id_categoria integer [ref: - categoria.id]

}


table marcas {
  id integer [pk]
  marca string
  detalle varchar[250]
}


table categoria_marcas {
  id_categoria integer [ ref: > categoria.id ]
  id_marcas integer [ref: > marcas.id]

}

table carrito {
 id integer [pk]
 rut_usuario integer [ref: > usuario.rut_usuario]
 satus_carrito varchar[250]
 creacion_date datetime
}

table coment_byn {
  id integer [pk]
  rut_usuario integer [ref: > usuario.rut_usuario]
  id_articulo integer [ref: > articulobyn.id]
  publicacion_date datetime
  contenido varchar[250]

}

table articulobyn {
id integer [pk]
rut_usuario integer [ref: > usuario.rut_usuario]
titulo varchar[250]
contenido varchar[250]
publicacion_date datetime
autor varchar[250]
}

id integer [pk]

}
